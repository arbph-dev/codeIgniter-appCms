# Threejs

 une architecture en trois couches :

- **3js.js** : bootstrap
- **ThreeManager** : gestionnaire d'instances
- **ThreeScene** : une scène autonome
- classes annexes (`TerranGenerator`, `util`, ...)

Je ne repartirais **surtout pas de zéro**. Je préfère réutiliser tout ce qui est bon et l'adapter à l'architecture CMS.

## Ce que je conserverais quasiment tel quel

 `ThreeManager`,  `ThreeScene` , `TerranGenerator`, `util.js` sont correctement séparés.

En revanche, je changerais la manière dont ils sont instanciés.

Aujourd'hui :

```
DOM
   ↓
ThreeManager scanne .cp_threejs
   ↓
new ThreeScene(node)
```

Demain :

```
CMS
   ↓
ThreeRenderer.php
   ↓
<div class="cp_threejs"
     data-scene="cube"
     data-width="800"
     data-height="600">
```

puis

```
bootstrap.js
      ↓
initThree()
      ↓
ThreeManager
      ↓
ThreeScene
```

On retrouve exactement la philosophie du composant Leaflet.

import * as THREE from 'three';

// js/ihm/3js/terran_generator
export class TerranGenerator {
    #W; #H; #U; #V; #G; #M; #O;
    arrMARKS;

    constructor(TW = 240, TH = 120, VW = 20, VH = 20) {
        this.#W = TW;
        this.#H = TH;
        this.#U = VW;
        this.#V = VH;
        this.#G = new THREE.PlaneGeometry(this.#W, this.#H, this.#U, this.#V);
        this.#M = UTIL.getMaterialFromTextures(1);
        this.arrMARKS = [];
    }

    fillPlanar(ZMAX = 0, ZMIN = 5) {
        let idxPG = 0;
        for (let iH = 0; iH <= this.#V; iH++) {
            for (let iW = 0; iW <= this.#U; iW++) {
                idxPG = (((this.#U + 1) * iH + iW) * 3) + 2;
                this.#G.attributes.position.array[idxPG] = UTIL.getRandomArbitrary(ZMIN, ZMAX);
            }
        }
    }

    addMarks(S) {
        const SB = 0.5;
        const SB2 = SB / 2;
        const DW = this.#W / this.#U;
        const DH = this.#H / this.#V;
        const TW2 = this.#W / 2;
        const TH2 = this.#H / 2;

        for (let iH = 0; iH <= this.#V; iH++) {
            for (let iW = 0; iW <= this.#U; iW++) {
                const material = new THREE.MeshStandardMaterial({ color: "red", metalness: 0.5, roughness: 1 });
                let geo = new THREE.BoxGeometry(SB, SB, SB);
                let BOX = new THREE.Mesh(geo, material);
                this.arrMARKS.push(BOX);
                S.add(BOX);
                BOX.position.x = (iW * DW) - TW2 + SB2;
                BOX.position.y = 0;
                BOX.position.z = (iH * DH) - TH2 + SB2;
            }
        }
    }

    setPlaneColor(C = 0xff0f0f) {
        this.#M.color.setHex(C);
    }

    getMesh() {
        this.#G.computeVertexNormals();
        this.#O = new THREE.Mesh(this.#G, this.#M);
        return this.#O;
    }

    getMaterial() { return this.#M; }

    setMaterial(M) { this.#M = M; }
}

// js/ihm/3js/ThreeManager.js
//
// Gestionnaire de toutes les instances ThreeScene.
// Pattern calqué sur CarouselManager.js.
//
// ── Events bus disponibles ────────────────────────────────────────────────────
//
//   bus.publish('threejs:start',   'THREE_1')           → démarre une scène
//   bus.publish('threejs:stop',    'THREE_1')           → pause une scène
//   bus.publish('threejs:destroy', 'THREE_1')           → détruit une scène
//   bus.publish('threejs:scene',   { id:'THREE_1', type:'galaxy' }) → change de scène
//   bus.publish('threejs:start:all')                    → démarre toutes
//   bus.publish('threejs:stop:all')                     → pause toutes
//   bus.publish('threejs:list')                         → log debug
//
// ─────────────────────────────────────────────────────────────────────────────

import { bus }        from '../../core/eventBus.js'
import { ThreeScene } from './ThreeScene.js'

class ThreeManager {

    constructor() {
        this.scenes = new Map()      // id → ThreeScene
        this._registerEvents()
    }

    // ─── Init ─────────────────────────────────────────────────────────────────

    /**
     * Scanne le DOM, crée une ThreeScene par nœud .cp_threejs et la démarre.
     * Appelé une seule fois depuis initThreejs().
     */
    init() {
        const nodes = document.querySelectorAll('.cp_threejs')

        nodes.forEach(node => {
            const instance = new ThreeScene(node)
            this.scenes.set(instance.id, instance)
            instance.start()
        })

        console.log(`ThreeManager: ${this.scenes.size} scène(s) initialisée(s)`)
    }

    // ─── Accès ────────────────────────────────────────────────────────────────

    getById(id) {
        return this.scenes.get(id) ?? null
    }

    destroy(id) {
        const scene = this.getById(id)
        if (!scene) return
        scene.destroy()
        this.scenes.delete(id)
    }

    // ─── Bus Events ──────────────────────────────────────────────────────────

    _registerEvents() {

        // ── Contrôle individuel ───────────────────────────────────────────
        bus.subscribe('threejs:start', (id) => {
            this.getById(id)?.start()
        })

        bus.subscribe('threejs:stop', (id) => {
            this.getById(id)?.stop()
        })

        bus.subscribe('threejs:destroy', (id) => {
            this.destroy(id)
        })

        // ── Changement de type de scène à chaud ───────────────────────────
        // bus.publish('threejs:scene', { id: 'THREE_1', type: 'galaxy' })
        bus.subscribe('threejs:scene', ({ id, type }) => {
            const scene = this.getById(id)
            if (!scene) return
            scene.stop()
            scene.destroy()
            this.scenes.delete(id)

            // Recrée la scène avec le nouveau type
            const node = document.getElementById(id)
            if (!node) return
            node.dataset.scene = type
            const fresh = new ThreeScene(node)
            this.scenes.set(fresh.id, fresh)
            fresh.start()
            console.log(`ThreeManager: scène [${id}] recréée en mode "${type}"`)
        })

        // ── Actions globales ──────────────────────────────────────────────
        bus.subscribe('threejs:start:all', () => {
            this.scenes.forEach(s => s.start())
        })

        bus.subscribe('threejs:stop:all', () => {
            this.scenes.forEach(s => s.stop())
        })

        bus.subscribe('threejs:destroy:all', () => {
            this.scenes.forEach((s, id) => s.destroy())
            this.scenes.clear()
        })

        // ── Debug ──────────────────────────────────────────────────────────
        bus.subscribe('threejs:list', () => {
            console.log('ThreeManager — scènes actives :', [...this.scenes.keys()])
        })
    }
}

export const threeManager = new ThreeManager()

// js/ihm/3js/ThreeScene.js
//
// Une instance Three.js autonome attachée à un nœud DOM.
// Pattern calqué sur Carousel.js.
//
// Usage HTML :
//   <div id="THREE_1" class="cp_threejs"
//        data-scene="cube"
//        data-width="800"
//        data-height="600">
//   </div>
//
// data-scene : "cube" | "terrain" | "model" | "galaxy"
// data-model : chemin vers un .obj  (optionnel, pour data-scene="model")

import * as THREE from 'three';
import { OrbitControls }  from 'three/addons/controls/OrbitControls.js';
import { OBJLoader }      from 'three/addons/loaders/OBJLoader.js';
import { TerranGenerator } from './terran_generator.js';
import * as UTIL           from './util.js';

export class ThreeScene {

    constructor(rootNode) {
        this.root   = rootNode
        this.id     = rootNode.id
        this.type   = rootNode.dataset.scene  ?? 'cube'
        this.width  = parseInt(rootNode.dataset.width)  || 800
        this.height = parseInt(rootNode.dataset.height) || 600

        // ── Three.js core ──────────────────────────────────────────────────
        this.scene     = null
        this.camera    = null
        this.renderer  = null
        this.controls  = null
        this.clock     = null
        this.raycaster = null

        // ── Objets de scène (référencés pour update / dispose) ─────────────
        this._cube    = null
        this._galaxy  = null
        this._model   = null   // chargement async
        this._terran  = null
        this._terranMesh = null

        // ── Boucle rAF ────────────────────────────────────────────────────
        this._rafId   = null

        // ── IntersectionObserver — pause auto hors écran ───────────────────
        this._observer = null
        this._visible  = false

        // ── Bind ──────────────────────────────────────────────────────────
        this._animate  = this._animate.bind(this)
        this._onResize = this._onResize.bind(this)

        this._build()
        this._initObserver()
    }

    // ─── Construction ─────────────────────────────────────────────────────────

    _build() {
        // Scène
        this.scene = new THREE.Scene()

        // Caméra
        this.camera = new THREE.PerspectiveCamera(75, this.width / this.height, 0.1, 1000)
        this.camera.position.z = 5

        // Renderer
        this.renderer = new THREE.WebGLRenderer({ antialias: true })
        this.renderer.setSize(this.width, this.height)
        this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
        this.root.appendChild(this.renderer.domElement)

        // Contrôles orbitaux
        this.controls = new OrbitControls(this.camera, this.renderer.domElement)
        this.controls.enableDamping = true

        // Utilitaires
        this.clock     = new THREE.Clock()
        this.raycaster = new THREE.Raycaster()

        // Lumières communes
        this._addLights()

        // Repère axes (peut être désactivé via data-axes="false")
        if (this.root.dataset.axes !== 'false') {
            this.scene.add(new THREE.AxesHelper(5))
        }

        // Contenu selon type
        this._buildScene()

        // Resize responsive
        window.addEventListener('resize', this._onResize)
    }

    _addLights() {
        const ambient  = new THREE.AmbientLight(0xffffff, 1.5)
        const dirLight = new THREE.DirectionalLight(0xffffff, 1.0)
        dirLight.position.set(5, 10, 7)
        this.scene.add(ambient, dirLight)
    }

    _buildScene() {
        const builders = {
            cube    : () => this._buildCube(),
            terrain : () => this._buildTerrain(),
            model   : () => this._buildModel(),
            galaxy  : () => this._buildGalaxy(),
        }
        ;(builders[this.type] ?? builders.cube)()
    }

    // ─── Types de scènes ──────────────────────────────────────────────────────

    _buildCube() {
        const geo = new THREE.BoxGeometry(1.5, 1.5, 1.5)
        const mat = new THREE.MeshStandardMaterial({ color: 0x00aaff })
        this._cube = new THREE.Mesh(geo, mat)
        this.scene.add(this._cube)
    }

    _buildTerrain() {
        this._terran     = new TerranGenerator(60, 60, 30, 30)
        this._terran.fillPlanar(2, -2)
        this._terranMesh = this._terran.getMesh()
        this._terranMesh.rotateX(-Math.PI / 2)
        this.scene.add(this._terranMesh)
        this.camera.position.set(0, 20, 30)
        this.camera.lookAt(0, 0, 0)
    }

    _buildModel() {
        const path = this.root.dataset.model
            ?? './assets/img/3js/model3d/bateaux/barco/Barco_obj/Barco.obj'

        const loader = new OBJLoader()
        loader.load(path, (obj) => {
            // Centre le modèle sur l'origine
            const box    = new THREE.Box3().setFromObject(obj)
            const center = box.getCenter(new THREE.Vector3())
            obj.position.sub(center)

            this._model = obj
            this.scene.add(obj)
            console.log(`ThreeScene [${this.id}] — modèle chargé`)
        })
    }

    _buildGalaxy() {
        const count     = 8000
        const positions = new Float32Array(count * 3)
        for (let i = 0; i < count * 3; i++) {
            positions[i] = (Math.random() - 0.5) * 30
        }
        const geo = new THREE.BufferGeometry()
        geo.setAttribute('position', new THREE.BufferAttribute(positions, 3))
        const mat = new THREE.PointsMaterial({ color: 0xaaddff, size: 0.05 })
        this._galaxy = new THREE.Points(geo, mat)
        this.scene.add(this._galaxy)
        this.camera.position.set(0, 5, 12)
    }

    // ─── Boucle d'animation ───────────────────────────────────────────────────

    _animate() {
        this._rafId = requestAnimationFrame(this._animate)

        // Pause GPU si la scène n'est pas visible (IntersectionObserver)
        if (!this._visible) return

        this.controls.update()                       // damping obligatoire
        this._updateScene(this.clock.getDelta())
        this.renderer.render(this.scene, this.camera)
    }

    _updateScene(delta) {
        switch (this.type) {
            case 'cube':
                if (this._cube) {
                    this._cube.rotation.x += delta * 0.8
                    this._cube.rotation.y += delta * 0.5
                }
                break
            case 'galaxy':
                if (this._galaxy) this._galaxy.rotation.y += delta * 0.05
                break
            case 'model':
                if (this._model)  this._model.rotation.y  += delta * 0.3
                break
            case 'terrain':
                // terrain statique — la caméra orbit suffit
                break
        }
    }

    // ─── IntersectionObserver ─────────────────────────────────────────────────

    _initObserver() {
        this._observer = new IntersectionObserver(
            (entries) => {
                this._visible = entries[0].isIntersecting
                // Reset clock au retour en visibilité pour éviter un delta explosif
                if (this._visible) this.clock.getDelta()
            },
            { threshold: 0.1 }
        )
        this._observer.observe(this.root)
    }

    // ─── Resize ───────────────────────────────────────────────────────────────

    _onResize() {
        const w = this.root.clientWidth  || this.width
        const h = this.root.clientHeight || this.height
        this.camera.aspect = w / h
        this.camera.updateProjectionMatrix()
        this.renderer.setSize(w, h)
    }

    // ─── API publique ─────────────────────────────────────────────────────────

    /**
     * Lance la boucle de rendu.
     * Sans effet si déjà démarrée.
     */
    start() {
        if (this._rafId !== null) return
        this.clock.start()
        this._animate()
        console.log(`ThreeScene [${this.id}] — start`)
    }

    /**
     * Stoppe la boucle de rendu (pause, libère le GPU).
     */
    stop() {
        if (this._rafId !== null) {
            cancelAnimationFrame(this._rafId)
            this._rafId = null
        }
        this.clock.stop()
        console.log(`ThreeScene [${this.id}] — stop`)
    }

    /**
     * Détruit proprement : arrête la boucle, déconnecte l'observer,
     * dispose toutes les ressources GPU, retire le canvas du DOM.
     */
    destroy() {
        this.stop()

        window.removeEventListener('resize', this._onResize)

        if (this._observer) {
            this._observer.disconnect()
            this._observer = null
        }

        // Libération mémoire GPU (géométries + matériaux)
        this.scene.traverse((obj) => {
            if (obj.geometry) obj.geometry.dispose()
            if (obj.material) {
                const mats = Array.isArray(obj.material) ? obj.material : [obj.material]
                mats.forEach(m => m.dispose())
            }
        })

        this.renderer.dispose()

        if (this.renderer.domElement.parentNode === this.root) {
            this.root.removeChild(this.renderer.domElement)
        }

        console.log(`ThreeScene [${this.id}] — destroyed`)
    }
}

// js/ihm/3js/BaseScene.js
//
// Classe de base abstraite pour toutes les scènes Three.js.
// Gère : renderer, camera, OrbitControls, IntersectionObserver, resize, boucle rAF.
// Les sous-classes implémentent : _buildScene() et _updateScene(delta).
//
// Cycle de vie :
//   const scene = new MaScene(rootNode)   → _build() appelé dans le constructeur
//   scene.start()                          → démarre la boucle rAF
//   scene.stop()                           → pause (GPU libéré)
//   scene.destroy()                        → libère tout (GPU + DOM + events)

import * as THREE            from 'three';
import { OrbitControls }     from 'three/addons/controls/OrbitControls.js';
import { SceneTimer }        from './SceneTimer.js';
import * as SU               from './SceneUtils.js';

export class BaseScene {

    /**
     * @param {HTMLElement} rootNode  Nœud DOM cible (recevra le <canvas>).
     */
    constructor(rootNode) {
        if (new.target === BaseScene) {
            throw new Error('BaseScene est abstraite — utilisez une sous-classe.')
        }

        this.root   = rootNode
        this.id     = rootNode.id
        this.width  = parseInt(rootNode.dataset.width)  || 800
        this.height = parseInt(rootNode.dataset.height) || 600

        // ── Three.js core ──────────────────────────────────────────────────
        this.scene    = null
        this.camera   = null
        this.renderer = null
        this.controls = null
        this.timer    = new SceneTimer()

        // ── Boucle rAF ────────────────────────────────────────────────────
        this._rafId   = null

        // ── IntersectionObserver ─────────────────────────────────────────
        this._observer = null
        this._visible  = false

        // ── Bind ──────────────────────────────────────────────────────────
        this._animate  = this._animate.bind(this)
        this._onResize = this._onResize.bind(this)

        this._initCore()
        this._buildScene()       // implémentée par la sous-classe
        this._initObserver()
    }

    // ─── Initialisation du cœur Three.js ─────────────────────────────────────

    _initCore() {
        // Scène
        this.scene = new THREE.Scene()

        // Caméra
        this.camera = new THREE.PerspectiveCamera(75, this.width / this.height, 0.1, 2000)
        this.camera.position.set(0, 2, 5)

        // Renderer
        this.renderer = new THREE.WebGLRenderer({ antialias: true })
        this.renderer.setSize(this.width, this.height)
        this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
        this.root.appendChild(this.renderer.domElement)

        // OrbitControls
        this.controls = new OrbitControls(this.camera, this.renderer.domElement)
        this.controls.enableDamping = true

        // Resize
        window.addEventListener('resize', this._onResize)
    }

    // ─── IntersectionObserver ─────────────────────────────────────────────────

    _initObserver() {
        this._observer = new IntersectionObserver(
            (entries) => {
                this._visible = entries[0].isIntersecting
                if (this._visible) {
                    // Évite un delta explosif au retour en visibilité
                    this.timer.resume()
                }
            },
            { threshold: 0.1 }
        )
        this._observer.observe(this.root)
    }

    // ─── Boucle d'animation ───────────────────────────────────────────────────

    _animate() {
        this._rafId = requestAnimationFrame(this._animate)

        const delta = this.timer.tick()

        if (!this._visible) return         // hors viewport → pas de rendu GPU

        this.controls.update()             // damping nécessite un appel par frame
        this._updateScene(delta)           // implémentée par la sous-classe
        this.renderer.render(this.scene, this.camera)
    }

    // ─── Resize ───────────────────────────────────────────────────────────────

    _onResize() {
        const w = this.root.clientWidth  || this.width
        const h = this.root.clientHeight || this.height
        if (w === 0 || h === 0) return

        this.camera.aspect = w / h
        this.camera.updateProjectionMatrix()
        this.renderer.setSize(w, h)
    }

    // ─── Méthodes abstraites (à surcharger) ───────────────────────────────────

    /**
     * Construit le contenu de la scène.
     * Appelé une seule fois dans le constructeur.
     * @abstract
     */
    _buildScene() {
        throw new Error(`${this.constructor.name} doit implémenter _buildScene()`)
    }

    /**
     * Mise à jour par frame.
     * @abstract
     * @param {number} delta  Secondes écoulées depuis le frame précédent (scalé).
     */
    _updateScene(delta) {
        throw new Error(`${this.constructor.name} doit implémenter _updateScene(delta)`)
    }

    // ─── API publique ─────────────────────────────────────────────────────────

    /** Démarre la boucle de rendu. Sans effet si déjà active. */
    start() {
        if (this._rafId !== null) return
        this.timer.start()
        this._animate()
        console.log(`[${this.id}] start`)
    }

    /** Pause — arrête le rAF, libère le GPU. */
    stop() {
        if (this._rafId !== null) {
            cancelAnimationFrame(this._rafId)
            this._rafId = null
        }
        this.timer.stop()
        console.log(`[${this.id}] stop`)
    }

    /**
     * Libère toutes les ressources (GPU, DOM, events, observer).
     * À appeler avant de retirer le nœud du DOM ou de recréer la scène.
     */
    destroy() {
        this.stop()

        window.removeEventListener('resize', this._onResize)

        if (this._observer) {
            this._observer.disconnect()
            this._observer = null
        }

        // Dispose GPU
        this.scene.traverse((obj) => {
            if (obj.geometry) obj.geometry.dispose()
            if (obj.material) {
                const mats = Array.isArray(obj.material) ? obj.material : [obj.material]
                mats.forEach(m => m.dispose())
            }
        })
        this.renderer.dispose()

        if (this.renderer.domElement.parentNode === this.root) {
            this.root.removeChild(this.renderer.domElement)
        }

        console.log(`[${this.id}] destroyed`)
    }
}

// js/ihm/3js/SceneTimer.js
//
// Remplace THREE.Clock (deprecated depuis r168).
// Basé sur performance.now() — résolution sub-milliseconde.
//
// Fonctionnement :
//   const timer = new SceneTimer()
//   timer.start()
//
//   // dans la boucle rAF :
//   const delta = timer.tick()   // secondes écoulées depuis le tick précédent (scalées)
//   const t     = timer.elapsed  // secondes écoulées depuis start (scalées)
//
// Facteur temps :
//   timer.scale = 1    → temps réel  (défaut)
//   timer.scale = 2    → ralenti ×2   (simulation plus lente que le réel)
//   timer.scale = 0.5  → accéléré ×2  (simulation plus rapide que le réel)
//   convention : delta_simulé = delta_réel / scale  →  scale ↑ = ralenti ↑

export class SceneTimer {

    constructor() {
        this._t0      = null    // timestamp démarrage  (ms, performance.now)
        this._tLast   = null    // timestamp dernier tick (ms)
        this._delta   = 0       // secondes entre tick(n-1) et tick(n), scalé
        this._elapsed = 0       // secondes depuis start(), scalé
        this._running = false
        this._scale   = 1       // facteur d'échelle temps (>1 = ralenti)
    }

    // ─── Contrôle ─────────────────────────────────────────────────────────────

    start() {
        const now     = performance.now()
        this._t0      = now
        this._tLast   = now
        this._delta   = 0
        this._elapsed = 0
        this._running = true
        return this
    }

    stop() {
        this._running = false
        return this
    }

    /** Reprend sans remettre elapsed à zéro. */
    resume() {
        this._tLast = performance.now()  // évite un delta explosif au retour
        this._running = true
        return this
    }

    /** Remet elapsed à 0 sans stopper (rebouclage animation). */
    resetElapsed() {
        this._elapsed = 0
        return this
    }

    // ─── Tick ─────────────────────────────────────────────────────────────────

    /**
     * À appeler UNE FOIS par frame (début de la boucle animate).
     * @returns {number} delta en secondes (scalé)
     */
    tick() {
        if (!this._running) {
            this._delta = 0
            return 0
        }

        const now         = performance.now()
        const rawDelta    = (now - this._tLast) / 1000   // secondes réelles
        this._tLast       = now

        // Clamp delta brut — évite un saut lors d'un retour d'onglet inactif
        const clampedRaw  = Math.min(rawDelta, 0.1)

        this._delta   = clampedRaw / this._scale
        this._elapsed += this._delta

        return this._delta
    }

    // ─── Accesseurs ───────────────────────────────────────────────────────────

    /** Dernier delta (secondes, scalé). */
    get delta()   { return this._delta   }

    /** Temps écoulé depuis start (secondes, scalé). */
    get elapsed() { return this._elapsed }

    /** Facteur d'échelle temps. >1 = ralenti. */
    get scale()   { return this._scale }
    set scale(v)  { this._scale = Math.max(0.001, Number(v)) }

    // ─── Utilitaires ──────────────────────────────────────────────────────────

    /** Temps réel brut depuis start (ms). Utile pour le debug. */
    get elapsedRealMs() {
        return this._t0 !== null ? performance.now() - this._t0 : 0
    }

    toString() {
        return `SceneTimer | elapsed=${this._elapsed.toFixed(3)}s | delta=${(this._delta*1000).toFixed(2)}ms | scale=${this._scale}`
    }
}

                break

                break
            case 'terrain':
                break
        }
    }

// js/ihm/3js/TrainScene.js
//
// Scène pédagogique : croisement de deux trains.
// Hérite de BaseScene (renderer, camera, controls, timer, observer, resize).
//
// ── Physique ──────────────────────────────────────────────────────────────────
//
//   Espace réel  : PtA = 0 m   →   PtB = 100 000 m  (100 km)
//   Espace scène : PtA = 0     →   PtB = 1000 unités
//   scD = 100 (1 unité scène = 100 m réels)
//
//   Train 1 : +100 km/h (gauche → droite)
//   Train 2 : −150 km/h (droite → gauche)
//
//   Temps croisement réel :
//     t_cross = D / (|V1| + |V2|) = 100 000 / (100 000/3.6 + 150 000/3.6)
//             = 100 000 * 3.6 / 250 000 ≈ 1.44 s  (en m/s)
//
//   Animation accélérée : scTRAFF = TR_max / ANIM_DURATION_S
//   Ralenti automatique à l'approche du croisement : scT ↑ quand |Δx| < SLOW_THRESHOLD
//
// ── HTML ──────────────────────────────────────────────────────────────────────
//   <div id="THREE_TRAIN" class="cp_threejs"
//        data-scene="train"
//        data-width="900" data-height="400"
//        data-v1="100"       <!-- km/h train 1, positif = gauche→droite -->
//        data-v2="-150"      <!-- km/h train 2, négatif = droite→gauche -->
//        data-distance="100" <!-- km, distance totale du trajet         -->
//        data-duration="60"> <!-- secondes d'animation avant rebouclage -->
//   </div>
//
// ── Bus events ────────────────────────────────────────────────────────────────
//   (gérés par ThreeManager : threejs:start / threejs:stop / threejs:destroy)
//   bus.publish('threejs:stop', 'THREE_TRAIN')
//   bus.publish('threejs:start', 'THREE_TRAIN')

import * as THREE        from 'three';
import { OBJLoader }     from 'three/addons/loaders/OBJLoader.js';
import { BaseScene }     from './BaseScene.js';
import * as SU           from './SceneUtils.js';

export class TrainScene extends BaseScene {

    constructor(rootNode) {
        super(rootNode)     // appelle _initCore() puis _buildScene()
    }

    // ─── Paramètres physique (lus depuis data-attributes ou valeurs par défaut) ──

    _readParams() {
        const ds = this.root.dataset

        // Vitesses en km/h → converties en m/s
        const v1_kmh =  parseFloat(ds.v1       ?? 100)
        const v2_kmh =  parseFloat(ds.v2       ?? -150)
        this._V1_ms  = v1_kmh * 1000 / 3600
        this._V2_ms  = v2_kmh * 1000 / 3600

        this._SCENE_UNITS = 1000
        this._scD = this._DIST_M / this._SCENE_UNITS   // m / unité scène

        const TR1 = this._DIST_M / Math.abs(this._V1_ms)
        const TR2 = this._DIST_M / Math.abs(this._V2_ms)

        this._scTRAFF = TR / this._ANIM_S

        this._scT  = 1

        console.log(`[${this.id}] TR=${TR.toFixed(1)}s  scTRAFF=${this._scTRAFF.toFixed(2)}  animDuration=${this._ANIM_S}s`)
    }

    _buildScene() {
        this._readParams()

        // Caméra positionnée pour voir toute la voie
        this.camera.position.set(500, 8, 40)
        this.camera.lookAt(500, 0, 0)

        // Lumières

        ground.position.set(500, -0.5, 0)
        this.scene.add(ground)

        if (this.root.dataset.axes !== 'false') {
            this.scene.add(SU.axesHelper(10))
        }

        this._buildBornes()

        this._train2 = this._buildTrain(0xff4422, 3, 0.8, 0.8)   // rouge

        const mat  = new THREE.MeshStandardMaterial({ color })
        return new THREE.Mesh(geo, mat)
    }

    _buildBornes() {
        const geoS  = new THREE.BoxGeometry(0.15, 0.15, 0.15)
        const matKm = new THREE.MeshBasicMaterial({ color: 0xffff00 })  // jaune tous les 10
        const matM  = new THREE.MeshBasicMaterial({ color: 0x00ff88 })  // vert chaque km

        for (let i = 0; i <= 100; i++) {
            const borne = new THREE.Mesh(geoS, (i % 10 === 0) ? matKm : matM)
            borne.position.set(i * (this._SCENE_UNITS / 100), -0.3, -2)
            this.scene.add(borne)
        }
    }

    _buildHUD() {
        this._hud = document.createElement('div')
        Object.assign(this._hud.style, {
            position   : 'absolute',
            top        : '4px',
            left       : '4px',
            color      : '#aef',
            fontFamily : 'monospace',
            fontSize   : '11px',
            background : 'rgba(0,0,0,0.45)',
            padding    : '4px 8px',
            borderRadius : '4px',
            pointerEvents : 'none',
        })
        // Le nœud root doit avoir position:relative pour que absolute fonctionne
        this.root.style.position = 'relative'
        this.root.appendChild(this._hud)
    }

    /** Charge un modèle OBJ/JSON et remplace le mesh cube si succès. */
    _loadModel(targetMesh, path, flipZ) {
        // Désactivé si le fichier n'existe pas — la scène fonctionne sans
        const loader = new OBJLoader()
        loader.load(
            path,
            (obj) => {
                SU.normalizeSize(obj, 3)
                SU.centerObject(obj)
                if (flipZ) obj.rotation.z = -Math.PI / 2
                else       obj.rotation.z =  Math.PI / 2

                // Remplace visuellement le cube par le modèle
                targetMesh.add(obj)
                console.log(`[${this.id}] modèle chargé : ${path}`)
            },
            undefined,
            () => { /* silencieux si absent */ }
        )
    }

    // ─── Mise à jour par frame ────────────────────────────────────────────────

    _updateScene(delta) {

        // ── Application du facteur temps à ce delta ──────────────────────
        // timer.scale est géré globalement ; ici on applique en plus scT (ralenti auto)
        const dt = delta / this._scT    // dt = secondes simulées scalées

        // ── Mise à jour des positions (unités scène) ──────────────────────
        // vPos += vitesse_m.s-1 * dt_s * accelAnim / scaleScene
        this._pos1 += this._V1_ms * dt * this._scTRAFF / this._scD
        this._pos2 += this._V2_ms * dt * this._scTRAFF / this._scD

        // ── Rebouclage ────────────────────────────────────────────────────
        if (this._pos1 > this._SCENE_UNITS || this._pos2 < 0) {
            this._pos1 = 0
            this._pos2 = this._SCENE_UNITS
            this._scT  = 1
        }

        // ── Ralenti automatique au croisement ─────────────────────────────
        const dx = Math.abs(this._pos1 - this._pos2)

        if (dx >= this._SLOW_THRESHOLD) {
            this._scT = 1                           // vitesse normale
        } else if (dx > 0.01) {
            // Ralenti progressif : scT augmente quand dx → 0
            // Courbe : scT = 1 + 1/(2·dx/threshold)
            this._scT = 1 + (1 / (2 * dx / this._SLOW_THRESHOLD))
            this._scT = Math.min(this._scT, 8192)  // plafond de sécurité
        }

        // ── Stop train 2 à 0 ─────────────────────────────────────────────
        if (this._pos2 < 0) this._pos2 = 0

        // ── Placement des meshes ──────────────────────────────────────────
        this._train1.position.set(this._pos1, 0.4, 0)
        this._train2.position.set(this._pos2, 0.4, 1)

        // ── Caméra — suit le train 1 ──────────────────────────────────────
        this.camera.position.x = this._train1.position.x - 0.5
        // (y et z restent fixes, controls gère le reste)

        // ── HUD ───────────────────────────────────────────────────────────
        this._updateHUD(dt)
    }

    _updateHUD(dt) {
        if (!this._hud) return
        const t = this.timer.elapsed

        const dx    = Math.abs(this._pos1 - this._pos2)
        const close = dx < this._SLOW_THRESHOLD

        this._hud.innerHTML =
            `<b style="color:${close ? '#f84' : '#aef'}">` +
            (close ? '⚠ CROISEMENT' : 'En route') + '</b><br>' +
            `t = ${t.toFixed(2)} s<br>` +
            `T1 pos = ${(this._pos1 * this._scD / 1000).toFixed(2)} km<br>` +
            `T2 pos = ${(this._pos2 * this._scD / 1000).toFixed(2)} km<br>` +
            `Δx = ${(dx * this._scD / 1000).toFixed(3)} km<br>` +
            `scT = ${this._scT.toFixed(1)} ${this._scT > 1 ? '🐢' : ''}`
    }

    // ─── Destroy (surcharge pour nettoyer le HUD) ─────────────────────────────

    destroy() {
        if (this._hud && this._hud.parentNode === this.root) {
            this.root.removeChild(this._hud)
        }
        super.destroy()
    }
}

// js/ihm/3js/Vector3D.js
//
// Primitive géométrique représentant un vecteur dans R³.
// Composition : sphere (origine) + cylinder (corps) + cone (pointe).
//
// ── Usage ────────────────────────────────────────────────────────────────────
//
//   import { Vector3D } from './Vector3D.js'
//
//   const v = new Vector3D({
//       label     : 'F',
//       origin    : new THREE.Vector3(0, 0, 0),
//       direction : new THREE.Vector3(1, 1, 0),   // sera normalisé
//       magnitude : 3,                             // longueur visuelle
//       color     : 0xff4422,
//   })
//
//   scene.add(v.group)          // ajouter à la scène Three.js
//   v.set(origin, dir, mag)     // mettre à jour dynamiquement
//   v.setColor(0x00aaff)
//   v.dispose()                 // libérer GPU
//
// ── Anatomie ─────────────────────────────────────────────────────────────────
//
//   group (THREE.Group)
//   ├── _sphere    (origine)          rayon = ORIGIN_R
//   ├── _shaft     (corps)            cylinder de longueur (mag - TIP_H)
//   └── _tip       (pointe flèche)    cone de hauteur TIP_H
//
//   L'axe de référence local est Y (0,1,0).
//   Le groupe est pivoté pour aligner Y sur la direction demandée.
//
// ── Constantes de proportion ──────────────────────────────────────────────────
//
//   ORIGIN_R   : rayon sphère origine
//   SHAFT_R    : rayon cylindre
//   TIP_R      : rayon base du cône
//   TIP_RATIO  : fraction de la magnitude consacrée à la pointe (0.18 → 18%)

import * as THREE from 'three';

const DEFAULTS = {
    label      : '',
    origin     : new THREE.Vector3(0, 0, 0),
    direction  : new THREE.Vector3(0, 1, 0),
    magnitude  : 2,
    color      : 0xffffff,
    showOrigin : true,    // afficher la sphère d'origine
    opacity    : 1.0,
}

const ORIGIN_R   = 0.07
const SHAFT_R    = 0.03
const TIP_R      = 0.10
const TIP_RATIO  = 0.22   // 22% de la magnitude pour la pointe
const SEG_R      = 12     // segments radiaux (performance vs qualité)
const SEG_H      = 1      // segments hauteur cylinder

// Axe local de référence (Three.js CylinderGeometry est aligné sur Y)
const AXIS_REF = new THREE.Vector3(0, 1, 0)
// Axe alternatif si direction ≈ -Y (évite quaternion NaN)
const AXIS_ALT = new THREE.Vector3(1, 0, 0)


export class Vector3D {

    /**
     * @param {object} params  Voir DEFAULTS ci-dessus.
     */
    constructor(params = {}) {
        this._p = { ...DEFAULTS, ...params }

        // Clones internes (évite mutation des objets passés en paramètre)
        this._origin    = this._p.origin.clone()
        this._direction = this._p.direction.clone().normalize()
        this._magnitude = this._p.magnitude
        this._color     = this._p.color

        this.label = this._p.label

        // ── Groupe Three.js racine ───────────────────────────────────────
        this.group = new THREE.Group()
        this.group.name = `vector_${this.label || 'unnamed'}`

        // ── Sous-meshes ──────────────────────────────────────────────────
        this._sphere = null
        this._shaft  = null
        this._tip    = null
        this._mat    = null     // matériau partagé

        this._build()
        this._applyTransform()
    }

    // ─── Construction initiale ────────────────────────────────────────────────

    _build() {
        this._mat = new THREE.MeshStandardMaterial({
            color       : this._color,
            roughness   : 0.4,
            metalness   : 0.2,
            transparent : this._p.opacity < 1,
            opacity     : this._p.opacity,
        })

        // Sphère origine
        this._sphere = new THREE.Mesh(
            new THREE.SphereGeometry(ORIGIN_R, SEG_R, SEG_R),
            this._mat
        )
        this._sphere.visible = this._p.showOrigin
        this._sphere.name = 'origin'

        // Corps (cylinder) — construit le long de Y local
        this._shaft = new THREE.Mesh(
            this._makeShaftGeo(),
            this._mat
        )
        this._shaft.name = 'shaft'

        // Pointe (cone) — construit le long de Y local
        this._tip = new THREE.Mesh(
            this._makeTipGeo(),
            this._mat
        )
        this._tip.name = 'tip'

        this.group.add(this._sphere, this._shaft, this._tip)
    }

    _makeShaftGeo() {
        const tipH   = this._magnitude * TIP_RATIO
        const shaftH = Math.max(this._magnitude - tipH, 0.001)
        const geo = new THREE.CylinderGeometry(SHAFT_R, SHAFT_R, shaftH, SEG_R, SEG_H)
        // On translate pour que la base du cylindre soit à Y=0 (origine)
        geo.translate(0, shaftH / 2, 0)
        return geo
    }

    _makeTipGeo() {
        const tipH   = this._magnitude * TIP_RATIO
        const shaftH = Math.max(this._magnitude - tipH, 0.001)
        const geo = new THREE.ConeGeometry(TIP_R, tipH, SEG_R)
        // Base du cône à Y = shaftH, pointe à Y = magnitude
        geo.translate(0, shaftH + tipH / 2, 0)
        return geo
    }

    // ─── Transformation (position + orientation) ──────────────────────────────

    _applyTransform() {
        // 1. Position du groupe = origine du vecteur
        this.group.position.copy(this._origin)

        // 2. Rotation : aligne Y local sur la direction demandée
        this.group.quaternion.copy(
            this._quaternionToDir(this._direction)
        )
    }

    _quaternionToDir(dir) {
        const d = dir.clone().normalize()
        // Cas dégénéré : direction = -Y → on passe par un axe intermédiaire
        if (d.dot(AXIS_REF) < -0.9999) {
            return new THREE.Quaternion().setFromAxisAngle(AXIS_ALT, Math.PI)
        }
        return new THREE.Quaternion().setFromUnitVectors(AXIS_REF, d)
    }

    // ─── Rebuild des géométries (magnitude changée) ───────────────────────────

    _rebuildGeometries() {
        // Dispose anciens GPU buffers
        this._shaft.geometry.dispose()
        this._tip.geometry.dispose()

        // Reconstruit
        this._shaft.geometry = this._makeShaftGeo()
        this._tip.geometry   = this._makeTipGeo()
    }

    // ─── API publique ─────────────────────────────────────────────────────────

    /**
     * Met à jour la position d'origine, la direction et la magnitude.
     * Les paramètres non fournis restent inchangés.
     *
     * @param {THREE.Vector3} [origin]
     * @param {THREE.Vector3} [direction]
     * @param {number}        [magnitude]
     */
    set(origin, direction, magnitude) {
        let needRebuild = false

        if (origin    !== undefined) this._origin    = origin.clone()
        if (direction !== undefined) this._direction = direction.clone().normalize()
        if (magnitude !== undefined && magnitude !== this._magnitude) {
            this._magnitude = Math.max(magnitude, 0.01)
            needRebuild = true
        }

        if (needRebuild) this._rebuildGeometries()
        this._applyTransform()
    }

    /**
     * Déplace uniquement l'origine (translation, sans rotation).
     * @param {THREE.Vector3} origin
     */
    setOrigin(origin) {
        this._origin = origin.clone()
        this.group.position.copy(this._origin)
    }

    /**
     * Change la direction (recalcule l'orientation).
     * @param {THREE.Vector3} direction
     */
    setDirection(direction) {
        this._direction = direction.clone().normalize()
        this.group.quaternion.copy(this._quaternionToDir(this._direction))
    }

    /**
     * Change la magnitude (reconstruit les géométries).
     * @param {number} magnitude
     */
    setMagnitude(magnitude) {
        this._magnitude = Math.max(magnitude, 0.01)
        this._rebuildGeometries()
    }

    /**
     * Change la couleur du vecteur entier.
     * @param {number} hexColor
     */
    setColor(hexColor) {
        this._color = hexColor
        this._mat.color.setHex(hexColor)
    }

    /**
     * Affiche/masque la sphère d'origine.
     */
    showOrigin(visible) {
        this._sphere.visible = visible
    }

    /** Retourne la position de la pointe (dans le repère monde). */
    getTip() {
        const localTip = new THREE.Vector3(0, this._magnitude, 0)
        return localTip.applyQuaternion(this.group.quaternion).add(this._origin)
    }

    /** Retourne le vecteur THREE.Vector3 complet (direction × magnitude). */
    toVector3() {
        return this._direction.clone().multiplyScalar(this._magnitude)
    }

    // ─── Accesseurs ───────────────────────────────────────────────────────────

    get origin()    { return this._origin.clone()    }
    get direction() { return this._direction.clone() }
    get magnitude() { return this._magnitude         }
    get color()     { return this._color             }

    // ─── Dispose ─────────────────────────────────────────────────────────────

    /**
     * Libère les ressources GPU.
     * Appeler avant de retirer le groupe de la scène.
     */
    dispose() {
        this._sphere.geometry.dispose()
        this._shaft.geometry.dispose()
        this._tip.geometry.dispose()
        this._mat.dispose()
        this.group.clear()
        console.log(`Vector3D [${this.label}] disposed`)
    }
}

        // ── Éléments de scène communs ──────────────────────────────────────
        SU.addDefaultLights(this.scene)
        this.scene.add(SU.grid(10, 10, 0x334455, 0x1a2233))
        this.scene.add(SU.axesHelper(3))
        this.scene.background = SU.skyboxColor(0x0d1117)

        // ── Labels CSS2D ───────────────────────────────────────────────────
        this._initCSS2D()

        // ── Bus events ─────────────────────────────────────────────────────
        this._registerVectorEvents()

        // ── Mode initial ───────────────────────────────────────────────────
        this._applyMode(this._mode)
    }

    // ─── CSS2DRenderer (labels flottants) ────────────────────────────────────

    _initCSS2D() {
        this._css2d = new CSS2DRenderer()
        this._css2d.setSize(this.width, this.height)

        Object.assign(this._css2d.domElement.style, {
            position      : 'absolute',
            top           : '0',
            left          : '0',
            pointerEvents : 'none',
        })
        this.root.style.position = 'relative'
        this.root.appendChild(this._css2d.domElement)
    }

    // ─── Modes pédagogiques ───────────────────────────────────────────────────

    _applyMode(mode) {
        // Nettoie tous les vecteurs existants
        this._clearAllVectors()
        this._mode = mode

        const builders = {
            demo  : () => this._buildDemo(),
            add   : () => this._buildAddition(),
            cross : () => this._buildCross(),
            field : () => this._buildField(),
        }
        ;(builders[mode] ?? builders.demo)()
    }

    // ── Mode DEMO : un vecteur qui tourne pour montrer la construction ─────────
    _buildDemo() {
        this.addVector('v1', {
            label     : 'v',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : new THREE.Vector3(1, 1, 0),
            magnitude : 3,
            color     : VCOL.BLUE,
        })
    }

    // ── Mode ADDITION : v1 + v2 = v3 ─────────────────────────────────────────
    _buildAddition() {
        const v1dir = new THREE.Vector3(2, 1, 0).normalize()
        const v2dir = new THREE.Vector3(0, 1, 1).normalize()

        this.addVector('v1', {
            label     : 'v₁',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : v1dir,
            magnitude : 2.5,
            color     : VCOL.RED,
        })

        this.addVector('v2', {
            label     : 'v₂',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : v2dir,
            magnitude : 2,
            color     : VCOL.GREEN,
        })

        this._updateSum()
    }

    // ── Mode PRODUIT VECTORIEL : v1 × v2 = v3 ────────────────────────────────
    _buildCross() {
        const v1 = new THREE.Vector3(1, 0, 0)
        const v2 = new THREE.Vector3(0, 1, 0)
        const v3 = new THREE.Vector3().crossVectors(v1, v2)

        this.addVector('v1', {
            label     : 'v₁',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : v1,
            magnitude : 2,
            color     : VCOL.RED,
        })
        this.addVector('v2', {
            label     : 'v₂',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : v2,
            magnitude : 2,
            color     : VCOL.GREEN,
        })
        this.addVector('cross', {
            label     : 'v₁×v₂',
            origin    : new THREE.Vector3(0, 0, 0),
            direction : v3,
            magnitude : 2,
            color     : VCOL.YELLOW,
        })
    }

    // ── Mode CHAMP : grille de vecteurs (préfigure E ou B) ────────────────────
    _buildField() {
        const N    = 5          // N×N vecteurs
        const step = 1.2
        const off  = (N - 1) * step / 2

        for (let ix = 0; ix < N; ix++) {
            for (let iz = 0; iz < N; iz++) {
                const ox = ix * step - off
                const oz = iz * step - off
                const origin = new THREE.Vector3(ox, 0, oz)

                // Exemple : champ radial centré (Coulomb 2D)
                const r   = origin.length() || 0.001
                const dir = origin.clone().normalize()
                const mag = Math.min(1 / (r * r) * 0.5, 1.2)

                this.addVector(`field_${ix}_${iz}`, {
                    label     : '',
                    origin,
                    direction : dir,
                    magnitude : mag,
                    color     : this._coulombColor(mag),
                    showOrigin: false,
                })
            }
        }
    }

    /** Couleur chaud→froid selon intensité. */
    _coulombColor(intensity, max = 1.2) {
        const t = Math.min(intensity / max, 1)
        const r = Math.round(255 * t)
        const b = Math.round(255 * (1 - t))
        return (r << 16) | b
    }

    // ─── Gestion de la collection de vecteurs ────────────────────────────────

    /**
     * Ajoute un vecteur à la scène.
     * @param {string} id      Identifiant unique.
     * @param {object} params  Paramètres Vector3D.
     * @returns {Vector3D}
     */
    addVector(id, params) {
        if (this._vectors.has(id)) this.removeVector(id)

        const v = new Vector3D(params)
        this.scene.add(v.group)
        this._vectors.set(id, v)

        // Label CSS2D
        if (params.label) {
            this._addLabel(v, params.label, params.color ?? 0xffffff)
        }

        return v
    }

    /**
     * Supprime et dispose un vecteur.
     */
    removeVector(id) {
        const v = this._vectors.get(id)
        if (!v) return
        this.scene.remove(v.group)
        v.dispose()
        this._vectors.delete(id)
    }

    /**
     * Retourne un vecteur par id.
     */
    getVector(id) { return this._vectors.get(id) ?? null }

    _clearAllVectors() {
        if (!this._vectors) return
        this._vectors.forEach((v, id) => this.removeVector(id))
    }

    // ─── Labels CSS2D ─────────────────────────────────────────────────────────

    _addLabel(vector3D, text, color) {
        const div = document.createElement('div')
        div.textContent = text

        Object.assign(div.style, {
            color      : `#${color.toString(16).padStart(6, '0')}`,
            fontFamily : 'serif',
            fontStyle  : 'italic',
            fontSize   : '16px',
            fontWeight : 'bold',
            textShadow : '0 0 6px rgba(0,0,0,0.9)',
            userSelect : 'none',
        })

        const label = new CSS2DObject(div)
        // Position : légèrement au-delà de la pointe
        label.position.set(0, vector3D.magnitude * 1.1, 0)
        vector3D.group.add(label)
    }

    // ─── Calcul d'addition (mis à jour en live) ───────────────────────────────

    _updateSum() {
        const v1 = this._vectors.get('v1')
        const v2 = this._vectors.get('v2')
        if (!v1 || !v2) return

        // r₃ = v1 + v2  (vecteurs complets)
        const r3 = new THREE.Vector3()
            .addVectors(v1.toVector3(), v2.toVector3())

        const mag3 = r3.length()
        const dir3 = r3.normalize()

        if (this._vectors.has('sum')) {
            this._vectors.get('sum').set(undefined, dir3, mag3)
        } else {
            this.addVector('sum', {
                label     : 'v₁+v₂',
                origin    : new THREE.Vector3(0, 0, 0),
                direction : dir3,
                magnitude : mag3,
                color     : VCOL.YELLOW,
            })
        }
    }

    // ─── Bus events ───────────────────────────────────────────────────────────

    _registerVectorEvents() {
        // bus.publish('vector:add', { sceneId:'THREE_VEC', id:'F', dir:[1,0,0], mag:2, color:0xff0000 })
        bus.subscribe('vector:add', (p) => {
            if (p.sceneId !== this.id) return
            this.addVector(p.id, {
                label     : p.id,
                origin    : p.origin    ? new THREE.Vector3(...p.origin)    : new THREE.Vector3(),
                direction : p.dir       ? new THREE.Vector3(...p.dir).normalize() : new THREE.Vector3(0,1,0),
                magnitude : p.mag       ?? 2,
                color     : p.color     ?? VCOL.WHITE,
            })
        })

        bus.subscribe('vector:remove', (p) => {
            if (p.sceneId !== this.id) return
            this.removeVector(p.id)
        })

        bus.subscribe('vector:update', (p) => {
            if (p.sceneId !== this.id) return
            const v = this.getVector(p.id)
            if (!v) return
            v.set(
                p.origin ? new THREE.Vector3(...p.origin) : undefined,
                p.dir    ? new THREE.Vector3(...p.dir)    : undefined,
                p.mag
            )
        })

        bus.subscribe('vector:setmode', (p) => {
            if (p.sceneId !== this.id) return
            this._applyMode(p.mode)
        })
    }

    // ─── Mise à jour par frame ────────────────────────────────────────────────

    _updateScene(delta) {
        this._time += delta

        switch (this._mode) {

            case 'demo': {
                // Rotation lente du vecteur démo autour de Y → montre l'orientation 3D
                const v = this._vectors.get('v1')
                if (!v) break
                const angle = this._time * 0.6
                v.setDirection(new THREE.Vector3(
                    Math.cos(angle),
                    0.6,
                    Math.sin(angle)
                ))
                break
            }

            case 'add': {
                // v2 oscille doucement → l'addition se recalcule en live
                const v2 = this._vectors.get('v2')
                if (!v2) break
                const a = this._time * 0.4
                v2.setDirection(new THREE.Vector3(
                    Math.sin(a) * 0.8,
                    1,
                    Math.cos(a) * 0.8
                ))
                this._updateSum()
                break
            }

            case 'cross': {
                // v1 tourne → v3 = v1×v2 se recalcule
                const v1 = this._vectors.get('v1')
                const v2 = this._vectors.get('v2')
                const vc = this._vectors.get('cross')
                if (!v1 || !v2 || !vc) break

                const a = this._time * 0.5
                v1.setDirection(new THREE.Vector3(Math.cos(a), 0, Math.sin(a)))

                const cross = new THREE.Vector3()
                    .crossVectors(v1.toVector3(), v2.toVector3())
                const mag = cross.length()
                if (mag > 0.001) {
                    vc.set(undefined, cross.normalize(), mag)
                }
                break
            }

            case 'field': {
                // Champ statique — pas d'update
                break
            }
        }

        // CSS2DRenderer — rendu des labels (en plus du WebGLRenderer)
        if (this._css2d) {
            this._css2d.render(this.scene, this.camera)
        }
    }

    // ─── Resize (surcharge — CSS2DRenderer à mettre à jour aussi) ─────────────

    _onResize() {
        super._onResize?.call(this)   // BaseScene
        const w = this.root.clientWidth  || this.width
        const h = this.root.clientHeight || this.height
        if (this._css2d) this._css2d.setSize(w, h)
    }

    // ─── Destroy (surcharge — CSS2DRenderer à nettoyer) ───────────────────────

    destroy() {
        this._clearAllVectors()
        if (this._css2d && this._css2d.domElement.parentNode === this.root) {
            this.root.removeChild(this._css2d.domElement)
        }
        super.destroy()
    }
}
